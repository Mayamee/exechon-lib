function [ANS] = fo(Q)
	q1 = Q(1);q2 = Q(2);q3 = Q(3);
	xA1 = -85;yA1 = 0;zA1 = 0;
	xA2 = 0;yA2 = 100;zA2 = 0;
	xA3 = 85;yA3 = 0;zA3 = 0;
	d_B1B3 = 130;de = 65;A1A3 = 170;
	opts = optimoptions(@fsolve,'Display','iter','MaxFunEvals',25000000, 'MaxIter',25000000,'TolFun',1E-10,'TolX',1E-10);
	% x0 = [xe ye ze k1 k2 k3 alf];
	x0 = [200 200 250 10 0.65 0.5 0.5];
	f = @(x)[
		(x(4)-d_B1B3*0.5*cos(x(6)))^2+(x(5)+d_B1B3*0.5*sin(x(6)))^2 - q1^2;
		(x(4)-xA2)^2+x(5)^2+de^2+yA2^2+zA2^2+2*(x(5)*yA2-de*zA2)*sin(x(7))-2*(de*yA2+x(5)*zA2)*cos(x(7)) - q2^2;
		(A1A3-(x(4)+d_B1B3*0.5*cos(x(6))))^2+(x(5)-d_B1B3*0.5*sin(x(6)))^2 - q3^2;
		atan(-x(2)/x(3)) - x(7);
		x(4)+xA1 - x(1);
		-x(5)*sin(x(7)) - x(2);
		x(5)*cos(x(7)) - x(3);
];
	SOL = fsolve(f,x0,opts);
    ANS = SOL;
end
